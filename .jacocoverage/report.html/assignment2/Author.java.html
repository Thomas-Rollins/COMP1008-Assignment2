    public Author(String firstName, String lastName, String nationality, LocalDate dateOfBirth) {
        
        setFirstName(firstName);
        setLastName(lastName);
        setNationality(nationality);
        setDateOfBirth(dateOfBirth);
        
    }

        return lastName;
    }
